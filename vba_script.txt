Sub stocks():'identifies last row in the fileDim last_row As Doublelast_row = Cells(Rows.Count, 1).End(xlUp).Row'creates active cell variableslast_row = Cells(Rows.Count, 1).End(xlUp).Rowactive_cell = ActiveCell.Value'sets volume attribute and value to zeroDim volume As Doublevolume = 0'loop to identify whether a given row matches the row beneath.  If a match occurs, the respective volume is added to the volume counterFor j = 2 To last_rowIf Cells(j, 1).Value = Cells(j + 1, 1).Value Thenvolume = (volume + Cells(j, 7).Value)'If a match does not occur, the volume is written to column j, the ticker symbol in column i, and the volume counter is resetElsevolume = (volume + Cells(j, 7).Value)Cells(j, 9).Value = Cells(j, 1).ValueCells(j, 10).Value = volumevolume = 0End IfNext j'insert row headers for columns i and jRange("i1") = "Ticker"Range("j1") = "Ticker Stock Volume"'removes blank cells in columns i and j and shifts cells upwardDim rng As Range    Set rng = Range("I2:j800000").SpecialCells(xlCellTypeBlanks)  rng.Rows.Delete Shift:=xlShiftUp  End Sub
